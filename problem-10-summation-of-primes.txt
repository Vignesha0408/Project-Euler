function isprime(n)
{
  if (n < 2) 
  return false;
  for(let i=2;i*i<=n;i++)
  if(n%i==0)
  return false;
  return true;
}
function primeSummation(n) {
  let sum=0;
  for(let i=2;i<n;i++)
  if(isprime(i))
    sum+=i;

  return sum;
}

primeSummation(2000000);
