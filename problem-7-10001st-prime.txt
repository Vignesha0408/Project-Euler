/*By listing the first six prime numbers: 2, 3, 5, 7, 11, and 13, we can see that the 6th prime is 13.

What is the nth prime number?
*/

function isprime(n){
  for(let i=2;i<n;i++)
    if(n%i==0)
      return false;
      return true;
}

function nthPrime(n) {
let count=0,prime=2,i=2;

  while(true)
  {
    if(isprime(i))
      {prime=i;
       count++; 
      }
    i++;
    if(count==n)
      return prime;
  }
  
}

nthPrime(10001);
